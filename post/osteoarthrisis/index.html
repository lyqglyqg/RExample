Step 1: Import data

Untar dataset into “GSE211584_raw/” folder using tar -xvf

library(stringr)
fs = list.files('./GSE211584_raw/',pattern = '^GSM')


samples=str_split(fs,'_',simplify = T)[,1]

lapply(unique(samples),function(x){
  y=fs[grepl(x,fs)]
  folder=paste0("GSE211584_raw/", str_split(y[1],'_',simplify = T)[,1])
  dir.create(folder,recursive = T)
  #为每个样本创建子文件夹
  file.rename(paste0("GSE211584_raw/",y[1]),file.path(folder,"barcodes.tsv.gz"))
  #重命名文件，并移动到相应的子文件夹里
  file.rename(paste0("GSE211584_raw/",y[2]),file.path(folder,"features.tsv.gz"))
  file.rename(paste0("GSE211584_raw/",y[3]),file.path(folder,"matrix.mtx.gz"))
})

dir='./GSE211584_raw'
sceList = lapply(samples,function(pro){ 
  #pro=samples2[2] 
  print(pro)  
  sce =CreateSeuratObject(counts =  Read10X(file.path(dir,pro)) ,
                          project =  pro  ,
                          min.cells = 5,

  return(sce)
})

sce.all=merge(x=sceList[[1]],

Step 2: Harmony, dimension reduction, and clustering

sce <- NormalizeData(sce, 
                         normalization.method = "LogNormalize",
                         scale.factor = 1e4) 
sce <- FindVariableFeatures(sce)
sce <- ScaleData(sce)
sce <- RunPCA(sce, features = VariableFeatures(object = sce))

library(harmony)
seuratObj <- RunHarmony(sce, "orig.ident")
names(seuratObj@reductions)
seuratObj <- RunUMAP(seuratObj,  dims = 1:15, 
                     reduction = "harmony")
DimPlot(seuratObj,reduction = "umap",label=T ) 

sce=seuratObj
sce <- FindNeighbors(sce, reduction = "harmony",
                     dims = 1:15) 
sce.all=sce
#设置不同的分辨率，观察分群效果(选择哪一个？)
